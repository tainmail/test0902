變數 匿名函式 調用 順序

田： 懶的記的話。先宣告函式，再調用就不會錯。

//正例 ==================================
fnA(); //一般函式，前後都在調用

function fnA(){
	alert("執行A")
};

fnA(); //一般函式，前後都在調用




//正例==================================
var fnB = function(){
	alert("執行B")
};

fnB(); //變數匿名函式，要在後面調用。




//反例==================================
try {
	fnC(); //變數匿名函式，不能在前面調用。
	var fnC = function(){
		alert("執行C")
	};
}catch(e){}
