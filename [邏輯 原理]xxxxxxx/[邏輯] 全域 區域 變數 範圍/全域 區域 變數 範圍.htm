全域 區域 變數 範圍

var sGlobal = "Global";  //宣告全域變數

function fnA(){
    var sLocal_a = "Local_a";  //宣告區域變數
    alert(sGlobal);  //全域變數，可用
    alert(sLocal_a); //同一層，區域變數，可用
    alert(sLocal_aa); //下一層，區域變數，不可用
    alert(sLocal_b); //不同區，區域變數，不可用
	
    function fnAA(){
        var sLocal_aa = "Local_aa";  //宣告區域變數
        alert(sGlobal);  //全域變數，可用
        alert(sLocal_a); //上一層，區域變數，可用
        alert(sLocal_aa); //同一層，區域變數，可用
        alert(sLocal_b); //不同區，區域變數，不可用
    };
};

function fnB(){
    var sLocal_b = "Local_b";  //宣告區域變數
    alert(sGlobal);  //全域變數，可用
    alert(sLocal_a); //不同區，區域變數，不可用
    alert(sLocal_aa); //不同區，區域變數，不可用
    alert(sLocal_b); //同一層，區域變數，可用
};
