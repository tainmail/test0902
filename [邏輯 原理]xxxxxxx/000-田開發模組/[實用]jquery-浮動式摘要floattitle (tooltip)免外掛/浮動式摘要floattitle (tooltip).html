/*浮動式摘要floattitle (tooltip)*/
.floattitle { width:300px; height:120px; border:1px solid red; border-radius:10px; 
background:#eee; position:absolute; z-index:10; opacity:0.9 }
.floattitle img { float:left; margin:5px; border:1px #fff solid }

<h3><a href="#" rel="images/pic_1s.jpg" title="安東尼真有義氣" >影韋德惡丟鞋舏瓜球砸臉回報</a></h3>

$(function(){
   var $oTarget = $("a")
   $oTarget.mouseover(function(e){
      this.sPicSrc = this.rel;
      this.sTitle = this.title;
      this.sTitle = (this.sTitle.length > 20 ? this.sTitle.toString().substring(0,20)+"..." : this.sTitle); //截字
      this.title = "";//清空原有title屬性值，防止瀏覽圖自帶的title
      var sHtml = "<div class='floattitle'>"+
      "<img src='"+this.sPicSrc+"'/><p>"+this.sTitle+"</p></div>";
      $('body').append(sHtml);
		
      $('.floattitle').css({
         top:(e.pageY+20),
         left:(e.pageX+10)
      }).show(200);
   });
	
	
   $oTarget.mouseout(function(){
      this.title = this.sTitle; //再放回title屬性值
      $('.floattitle').remove();
   });
	
   $oTarget.mousemove(function(e){
      $('.floattitle').css({
         top:(e.pageY+20),
         left:(e.pageX+10)
      });
   });
});
