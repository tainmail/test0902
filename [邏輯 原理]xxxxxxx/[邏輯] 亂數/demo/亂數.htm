亂數

Math.random()     //假設 0.485..    // 亂數無限循環小數




Math.random()*5   //假設 0.485 * 5 = 2.42      //5以下的小數


Math.round( Math.random()*5 )    //假設 0.485 * 5 = 2.42     //四捨五入輸出2        //5以下的整數
	



Math.round( Math.random()*(10-5) + 5 )     //假設 0.485 * (10-5) = 2.42     //10到5有五個數     //然後從5開始計
	



//寫成一個function
function fnRandom(argMin, argMax){
    return Math.round(argMin + Math.random()*(argMax - argMin)); 
};
	
fnRandom(5, 10)
