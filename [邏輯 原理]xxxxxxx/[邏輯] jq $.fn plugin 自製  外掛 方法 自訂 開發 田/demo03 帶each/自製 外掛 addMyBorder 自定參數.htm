自製外掛.addMyBorder()
本例。自訂預設參數
  
//寫入自訂的外掛==============
(function($) {
	$.fn.addMyBorder = function(options){
		
		//自訂預設參數
		var settings = $.extend({
			sBorder: "5px solid green",
			iWidth:100,
			iHeight:50
		}, options );
		
		return this.each(function(){
			$(this).css({
				border: settings.sBorder,
				width: settings.iWidth,
				height: settings.iHeight
			});
		});
	};
	
}(jQuery));



//調用===============
$(function(){
	
	$(".btn_11").click(function(){
		$(this).addMyBorder({
			sBorder: "5px solid orange",
			iWidth:50,
			iHeight:100
		});
	});
	
	$(".btn_12").click(function(){
		$(this).addMyBorder({
			sBorder: "5px dashed red",
			iWidth:100,
			iHeight:50
		});
	});
	
});
