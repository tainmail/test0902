變數++i 與 i++ 遞增 前置增值 後置增值
速記 
a++    //後置增值    a先回傳，再+1
a--    //後置減值    a先回傳，再-1

++a    //前置增值    a先+1，再回傳
--a    //前置減值    a先-1，再回傳
原理 
//後加加==================
var a;  
i=6;  

a=i++;   //將 a=i  所以 a=6     //最後 i+1  所以 i=7  

document.write(i);  //輸出 7
document.write(a);//輸出 6



//前加加=====================
var a;  
var i=6;    

a=++i;   // i+1 所以 i=7   //將a=i  所以a=7  

document.write(i);   //輸出 7
document.write(a);  //輸出 7
